def _gdb(my_gdb):












    if my_gdb == 1:












        os.system("echo \"source ~/pwndbg/gdbinit.py\nsource ~/Pwngdb/pwngdb.py\n"












                  "source ~/Pwngdb/angelheap/gdbinit.py\" > ~/.gdbinit")












        gdb.attach(taolve)












    elif my_gdb == 2:












        os.system("echo \"source ~/gef/gef.py\" > ~/.gdbinit")












        gdb.attach(taolve)












    elif my_gdb == 3:












        os.system("echo \"source ~/peda-heap/peda.py\" > ~/.gdbinit")












        gdb.attach(taolve)






































def _cmd(num):












    taolve.recvuntil(b'>')












    taolve.sendline(str(num).encode('utf-8'))






































def _add(idx, size):












    _cmd(1)












    taolve.sendlineafter(b'Index: ', tob(idx))












    taolve.sendlineafter(b'Size: ', tob(size))
