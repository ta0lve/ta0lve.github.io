1.6.3 反转链表 # 
#include <iostream>
using namespace std;
class ListNode{
public:
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(NULL) {}    
};
ListNode* reverseList(ListNode* head) 
{
    ListNode* prev = nullptr;  // 前驱节点指针
    ListNode* curr = head;     // 当前节点指针

    while (curr != nullptr) 
    {
        ListNode* next = curr->next;  // 后继节点指针
        curr->next = prev;                 // 反转箭头指向前驱节点
        prev = curr;                       // 更新前驱节点指针
        curr = next;                   // 更新当前节点指针
    }

    return prev;  // 返回反转后的头节点
}

int main() {
    // 构建一个链表，输入为 1 2 3 4 5
    ListNode* head = new ListNode(1);
    ListNode* curr = head;
    for (int i = 2; i <= 5; i++) {
        curr->next = new ListNode(i);
        curr = curr->next;
    }

    // 调用反转链表函数
    head = reverseList(head);

    // 输出反转后链表的值
    curr = head;
    while (curr != nullptr) {
        std::cout << curr->val << " ";
        curr = curr->next;
    }

    // 释放链表内存
    curr = head;
    while (curr != nullptr) {
        ListNode* temp = curr;
        curr = curr->next;
        delete temp;
    }

    return 0;
}

1.7.1 深拷贝 #

在进行深拷贝时，需要注意以下两点：

  1. 检查目标对象中的数据成员是否已经分配内存（是否不等于nullptr），如果是，则需要先释放（delete），防止内存泄漏
  2. 创建新的内存空间，并将原始数据拷贝到新空间中，确保每个对象拥有独立的内存副本
//一维数组深拷贝
class array
{
private:
    int size;  // size
    int *data; // dynamically allocate/release memory

public:
    array()；
    array(int size, int val);
    array(const array &another)
    {
        data = nullptr;
        //执行构造函数是还未分配内存空间，所以不用delete
        deep_copy(another);
    }
    ~array();
    void deep_copy(const array &another);
};

array::array()
{
    size = 0;
    data = nullptr;
}

array::array(int size, int val)
{
    this->size=size;
    this->data=NULL;
    data=new int[size];
    for(int i=0;i<size;i++)
    {
        data[i]=val;
    }
}

array::~array()
{
    delete [] data;
}

void array::deep_copy(const array &another)
{
    this->size=another.size;
    //检查data是否为nullptr
    if(data!=nullptr)
    {
        delete [] data;
    }
    //指向新创建的内存空间
    data=new int[this->size];
    //完成深拷贝
    for(int i=0;i<size;i++)
    {
        data[i]=another.data[i];
    }
}

//重载赋值运算符时也要记得先释放空间

1.7.2 写时拷贝 #

(以一维数组为例)

观察一维数组类的deep_copy函数的实现，我们发现，在每次进行深拷贝的时候，我们都需要重新分配一块新的内存。在很多时候，array的实例可能只需要调用print函数（不改变值的打印函数），不需要调用assign函数（改变值的赋值函数），即实例不会对data指向的内存进行修改。对于这些实例，data指向的内存所存放的内容是相同的，因此可以共享，不需要重新分配内存，从而达到减少内存开销的目的。但是，我们并不能提前预知每一个实例在使用过程中是否会对data指向的内存进行修改，不难想到，只需要将深拷贝的过程延迟到实例第一次发生修改的地方即可，这就是写时复制，copy on write。

写时复制的具体实现：

  • 成员变量ref_counter的作用是标记当前引用data指向的内存的实例数量。
  • 在非复制构造函数中，ref_counter指向的整型变量初始化为1。
  • 当调用deep_copy函数时，并不会执行深拷贝，而是执行浅拷贝，然后对ref_counter进行加1。
  • copy_on_write是真正执行深拷贝的函数。
  • 当需要释放data和ref_counter指向的内存时，如果ref_counter指向的值为1，则释放；否则，对ref_counter指向的值减1，不进行释放操作。
class array
{
private:
    int size;         // size
    int *data;        // dynamically allocate/release memory
    int *ref_counter; // reference counter

    void copy_on_write(); // copy on write

public:
    array();
    array(int size, int val);
    array(const array &another)
    {
        data = nullptr;
        ref_counter = nullptr;
        deep_copy(another);
    }
    ~array();
    bool assign(int pos, int val);
    void deep_copy(const array &another);
    void print() const
    {
        for (int i = 0; i < size; ++i)
            std::cout << data[i] << std::endl;
    }
    int get_ref_counter() { return *ref_counter; }
    int *get_data() { return data; }
};

#include<iostream>
#include "array.h"
array::array()
{
    size = 0;
    data = nullptr;
    ref_counter=new int;
    *ref_counter=1;
}

array::array(int size, int val)
{
    this->size = size;
    ref_counter=new int;
    *ref_counter=1;
    data = new int[size];
    for (int i = 0; i < size; ++i)
        data[i] = val;
}

array::~array()
{
    if((*ref_counter)==1)
    {
       delete ref_counter;
       delete[] data;
    }
    else
    (*ref_counter)--;   
}

// 如果pos不越界，则将pos位置的元素赋值为val，返回true；如果pos越界，则返回false 
bool array::assign(int pos, int val) {
    if (*ref_counter != 1) 
    {
        (* ref_counter)--;
        copy_on_write();
    }
    bool result = true; //标记是否成功
    if (pos >= 0 && pos < this->size) 
    {
        this->data[pos] = val;//在深拷贝后的新内存空间中改变data
    }
    else 
    {
        result = false;
    }
    return result;
}

//假的深拷贝函数
void array::deep_copy(const array& another) 
{
    if (&another != this) 
    {
        size = another.size;
        if (ref_counter != nullptr&& *ref_counter==1) 
        {
            delete[] data;
            delete ref_counter;
        }
        else if (ref_counter != nullptr && *ref_counter != 1) 
        {
            (*ref_counter)--;
        }
        ref_counter = another.ref_counter;
        (*ref_counter)++;
        data = another.data;
    }
}

//真正的深拷贝函数——写时拷贝函数
void array::copy_on_write() 
{
    ref_counter = new int;
    *ref_counter = 1;
    int* temp = data; //指向原本的data内存空间
    data = new int[size]; //创建一个新的内存空间（深拷贝）
    for (int i = 0; i < size; i++) 
    {
        data[i] = temp[i]; //完成深拷贝
    }
}

1.9.1 虚析构函数 #

我们知道，为了能够正确的调用对象的析构函数，一般要求具有层次结构的顶级类定义其析构函数为虚函数。因为在delete一个抽象类指针时候，必须要通过虚函数找到真正的析构函数。

如：

class Base  
{  
public:  
        Base(){}  
        virtual ~Base(){}  
};  

class Derived: public Base  
{  
public:  
   Derived(){};  
   ~Derived(){};  
}  

void foo()  
{  
   Base *pb;  
   pb = new Derived;  
   delete pb;  
}

这是正确的用法，会发生动态绑定，它会先调用Derived的析构函数，然后是Base的析构函数

如果析构函数不加virtual，delete pb只会执行Base的析构函数，而不是真正的Derived析构函数。

因为不是virtual函数，所以调用的函数依赖于指向静态类型，即Base。

参考链接： 析构函数可以是纯虚函数吗?

最后，总结一下关于虚函数的一些常见问题：

  1. 虚函数是动态绑定的，也就是说，使用虚函数的指针和引用能够正确找到实际类的对应函数，而不是执行定义类的函数。这是虚函数的基本功能，就不再解释了。
  2. 构造函数不能是虚函数。而且，在构造函数中调用虚函数，实际执行的是父类的对应函数，因为自己还没有构造好, 多态是被disable的。
  3. 析构函数可以是虚函数，而且，在一个复杂类结构中，这往往是必须的。
  4. 将一个函数定义为纯虚函数，实际上是将这个类定义为抽象类，不能实例化对象。
  5. 纯虚函数通常没有定义体，但也完全可以拥有, 甚至可以显示调用。
  6. 析构函数可以是纯虚的，但纯虚析构函数必须有定义体，*因为析构函数的调用是在子类中隐含的*。
  7. 非纯的虚函数必须有定义体，不然是一个错误。
  8. 派生类的override虚函数定义必须和父类完全一致(c++11中使用override进行编译器检查)。除了一个特例，如果父类中返回值是一个指针或引用，子类override时可以返回这个指针（或引用）的派生。例如，在上面的例子中，在Base中定义了virtual Base* clone(); 在Derived中可以定义为 virtual Derived* clone()。可以看到，这种放松对于Clone模式是非常有用的(也就是说override并不会检查返回值类型)。

/*栈 Stack 是一种先入后出的数据结构，最先入栈的元素称为栈底，最后入栈的元素称为栈顶。为了方便，可用 node.hpp 中的链表结构实现栈，并用链表头指向栈顶元素。

请实现一个类模板 Stack ，使其可以保存不同类型的数据。*/
#include "genericStack.hpp"
#include <iostream>
using namespace std;

int main() //程序EX6_4.cpp
{
    Stack<int> stack; // 实例化一个保存int型元素的栈
    for (int i = 1; i < 9; i++) // 向栈中压入8个元素
        stack.push(i);
    while (!stack.isEmpty())
    {                                  // 栈不为空时循环
        cout << stack.getTop() << " "; // 显示栈顶元素
        stack.pop();                   // 弹出栈顶元素
    }

}

 //genericStack.h
template <typename ElementType>
class Stack{
public:
  Stack()
  {
    top = nullptr;
  }
  ~Stack()
  {
    while(top){
        Node *temp = top;
        top = top->next;
        delete temp;
    }
  }
  void push(ElementType obj)
  {
    Node *newNode = new Node{obj, top};
    top = newNode;
  }
 //将新元素入栈
  void pop()
  {
    if(!isEmpty()){
        Node *temp = top;
        top = top->next;
        delete temp;
    }
  }                     //将栈顶元素出栈
  ElementType getTop() const
  {
    return top->element;
  }     //返回栈顶元素
  bool isEmpty() const
  {
    return top==nullptr;
  }           //判断栈是否为空

private:
  struct Node{           // 栈结点类型
    ElementType element; // 结点中存放的元素
    Node *next;	         // 指向下一结点的指针
  };
  Node *top;         // 栈顶
};
