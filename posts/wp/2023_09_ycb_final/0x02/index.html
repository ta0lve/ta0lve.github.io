main 函数 #

然后打开IDA开始分析

点击main函数，查看其伪代码，可以看到是一个菜单题

有4个功能：

int _menu()
{
  puts("1. Add note");
  puts("2. Delete note");
  puts("3. Edit note");
  puts("4. Show note");
  return putchar(62);

  • 一般情况下我们都会选择直接先运行一下二进制程序来了解一下程序的功能，不过这道题当我们运行时却提示我们：

    • ta0lve@ta0lve:~/桌面/CTF/ycb2023_final/pwn2$ ./pwn
No! you can't do this!

    • 所以我们先看看这个字符串在哪里出现吧

    • FILE *sub_40127C()
{
  FILE *result; // rax

  if ( *(FILE **)off_404070 != stderr
    || *(FILE **)off_404078 != stdout
    || (result = stdin, *(FILE **)off_404080 != stdin) )
  {
    write(2, "No! you can't do this!", 0x17uLL);
    _exit(0);
  }
  return result;

    • 简单的来讲就是程序检查了一下一些地址上面的函数是否与设定上的相同，而且1这些函数都是在libc中的

    • 所以我们patchelf二进制文件，用题目给我们的libc和ld来启动程序

$ patchelf --set-interpreter ./ld* ./pwn

add 函数 # 

unsigned __int64 __fastcall _add(const char *a1)
{
  unsigned int v1; // ebx
  unsigned int v3; // [rsp+0h] [rbp-20h] BYREF
  _DWORD size[5]; // [rsp+4h] [rbp-1Ch] BYREF

  *(_QWORD *)&size[1] = __readfsqword(0x28u);
  printf("Index: ");
  __isoc99_scanf("%u", &v3);
  if ( v3 <= 0xF )                              // 只能申请0x10个堆块
  {
    printf("Size: ");
    __isoc99_scanf("%u", size);
    if ( size[0] <= 0x900u )                    // 不能大于0x900
    {
      if ( size[0] > 0x4FFu )                   // 不能小于0x4FF,所以我们只能申请large bin
      {
        v1 = v3;
        chunk[v1] = malloc(size[0]);            // 指针存放在chunk数组里(rename from qword_4040E0)
        chunk_size[v3] = size[0];               // chunk的size也存放在一个数组里(rename from dword_404160)
      }
      else
      {
        puts("Too small.");
      }
    }
    else
    {
      puts("Too big.");
    }
  }
  else
  {
    puts("There are only 16 pages in this notebook.");
  }
  return *(_QWORD *)&size[1] - __readfsqword(0x28u);
}
  • 推测应该是Large Bin Attack

del 函数 # 

unsigned __int64 __fastcall _del(const char *a1)
{
  unsigned int v2; // [rsp+4h] [rbp-Ch] BYREF
  unsigned __int64 v3; // [rsp+8h] [rbp-8h]

  v3 = __readfsqword(0x28u);
  printf("Index: ");
  __isoc99_scanf("%u", &v2);
  if ( v2 <= 0xF )
  {
    if ( chunk[v2] )
      free((void *)chunk[v2]);                  // uaf
                                                // 可以泄露libc
                                                // fix的时候可以考虑patch这个地方
    else
      puts("Page not found.");
  }
  else
  {
    puts("There are only 16 pages in this notebook.");
  }
  return v3 - __readfsqword(0x28u);
}
  • 很明显有一个uaf漏洞，可以泄露出libc的基址

unsigned __int64 _show()
{
  unsigned int v1; // [rsp+4h] [rbp-Ch] BYREF

后门函数 #

其实这一步应该是打开IDA后做的第一件事：

shift + F12 查看程序的字符串，有时候字符串中会有一些语句帮助我们更快地找到关键代码，也有时候会有一些明显的 win 语句提示我们成功 getshell 或者 成功取得 flag，当然，最有价值的还得是/bin/sh/语句，这提示我们给程序可能存在后门函数，我们可以在之后的构造RO链、调用 system 函数时使用这个字符串。

当然，当我们在函数的引用表里看到 system 函数时应该也会想到类似的事情。

在本题中，

  • 双击 system 函数进入.plt.sec段，查看交叉引用XREF可以很快地找到后门函数
  • 双击/bin/sh字符串进入.rodata段，查看交叉引用XREF可以很快地找到后门函数
