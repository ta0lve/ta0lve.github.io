# task.py
from hashlib import sha256
import random
import string
import signal
from Crypto.Util.number import *


table = string.ascii_letters + string.digits


def proof_of_work():
    proof = (''.join([random.choice(table)for _ in range(12)])).encode()
    sha = sha256(proof).hexdigest()
    print("[+] sha256(XXXX+" + proof[4:].decode() + ") == " + sha)
    XXXX = input('[+] Plz Tell Me XXXX :').encode()
    if len(XXXX) != 4 or sha256(XXXX + proof[4:]).hexdigest() != sha:
        return False
    return True


def level1():
    p = getPrime(512)
    q = getPrime(512)
    n = p * q
    p_bin = bin(p)[2:]
    p_bin_corrupted = p_bin[:-24] + '█' * 24
    print(f'[+] n = {n}')
    print(f'[+] p = {p_bin_corrupted}')
    ans = input("[+] Now please tell me the last 24 bits of p: ")
    if ans == p_bin[-24:]:
        print('[+] Recovered successfully! Here is your flag:')
        print(open("/flag1", "r").read())
    else:
        print("[-] Wrong Answer!")


def level2():
    brokenbits = 114
    pqbits = 514
    p = getPrime(pqbits)
    q = getPrime(pqbits)
    n = p * q
    p_bin = bin(p)[2:]
    p_bin_corrupted = p_bin[:-brokenbits] + '█' * brokenbits
    print(f'[+] n = {n}')
    print(f'[+] p = {p_bin_corrupted}')
    ans = input("[+] Now please tell me the last 114 bits of p: ")
    if ans == p_bin[-brokenbits:]:
        print('[+] Recovered successfully! Here is your flag:')
        print(open("/flag2", "r").read())
    else:
        print("[-] Wrong Answer!")

分析 #

本题为环境题，题目分为三个部分：

  • 连过去之后首先是一个proof_of_work
    • 这是一个简单的工作量证明（Proof of Work）函数，它生成一个随机的长度为12的字符串，然后使用 SHA256 哈希算法计算哈希值。
    • 接着，通过用户输入一个4位的字符串 XXXX，将输入字符串与之前生成的随机字符串组合，并使用 SHA256 哈希算法计算哈希值。
    • 如果两个哈希值相等，且输入字符串长度为4，则返回 True，否则返回 False
  • 由于4位字符串XXXX并不长，我们可以采用枚举法的方式来枚举所有字符串并计算1其hash是否与题目所给的相同
  • 接下来就是两道Crypto题，分别是level1与level2，对应两个flag
  • level1被遮挡的数字brokenbits只有24位，也可以直接枚举爆破得出答案
  • level2brokenbits有114位，就需要我们通过数学的方法来得出被遮挡的数字了
